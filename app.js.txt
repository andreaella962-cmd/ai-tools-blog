fetch('articles.json')
  .then(res => res.json())
  .then(data => {
    const container = document.getElementById('articles');

    data.forEach(article => {
      const div = document.createElement('div');
      div.className = 'card';

      div.innerHTML = `
        <h2>${article.title}</h2>
        <p>${article.description}</p>
      `;

      container.appendChild(div);
    });
  });
